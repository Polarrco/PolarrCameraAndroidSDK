cmake_minimum_required(VERSION 3.4.1)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -fno-rtti -fno-exceptions -Wall")

add_library(PolarrRender SHARED IMPORTED)

set_target_properties(PolarrRender
                      PROPERTIES
                      IMPORTED_LOCATION
                      ${CMAKE_SOURCE_DIR}/jnis/${ANDROID_ABI}/libPolarrRender.so)


add_library(PolarrRenderJNI SHARED
            polarrRenderJni.cpp)

target_include_directories(PolarrRenderJNI PRIVATE
                           ${CMAKE_SOURCE_DIR}/include)

target_link_libraries(PolarrRenderJNI
                      PolarrRender
                      android
                      log)
